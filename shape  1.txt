import joblib
import shap
import numpy as np
import pandas as pd
import json

data   = joblib.load("RandomForest_model_and_scaler (1).pkl")
model  = data['model']
scaler = data['scaler']

explainer = shap.TreeExplainer(model)

def process_health_data_rf(raw_vitals_list):

    features = [
        'heart_rate',
        'blood_pressure',
        'oxygen_saturation',
        'respiratory_rate',
        'temperature'
    ]

    valid_ranges = {
        'heart_rate':        (40,  200),
        'blood_pressure':    (50,  250),
        'oxygen_saturation': (70,  100),
        'respiratory_rate':  (4,   60),
        'temperature':       (34,  42),
    }

    translations = {
        'heart_rate':        'Heart Rate',
        'blood_pressure':    'Blood Pressure',
        'oxygen_saturation': 'Oxygen Saturation',
        'respiratory_rate':  'Respiratory Rate',
        'temperature':       'Temperature',
    }

    # ── 1. Clip + collect warnings ──────────────────────────────
    clipped  = []
    warnings = []

    for val, feat in zip(raw_vitals_list, features):
        lo, hi = valid_ranges[feat]
        if not (lo <= val <= hi):
            warnings.append(
                f"{translations[feat]} value {val} is outside valid range ({lo}-{hi})"
            )
            val = np.clip(val, lo, hi)
        clipped.append(val)

    input_df    = pd.DataFrame([clipped], columns=features)
    scaled_data = scaler.transform(input_df)
    prediction  = int(model.predict(scaled_data)[0])

    shap_values = explainer.shap_values(scaled_data)

    if isinstance(shap_values, list):
       
        risk_values = np.array(shap_values[1])[0]
    elif shap_values.ndim == 3:
       
        risk_values = shap_values[0, :, 1]
    else:
        risk_values = shap_values[0]


    if prediction == 1:
       
        mask = risk_values > 0
        if not mask.any():
            mask = np.ones(len(risk_values), dtype=bool)
        top_indices = np.where(mask)[0]
        top_indices = top_indices[np.argsort(risk_values[top_indices])[::-1]][:3]

        explanation_text = "Risk detected: contributing factors are " + ", ".join(
            [translations[features[i]] for i in top_indices]
        )

    else:
        
        mask = risk_values < 0
        if not mask.any():
            mask = np.ones(len(risk_values), dtype=bool)
        top_indices = np.where(mask)[0]
        top_indices = top_indices[np.argsort(risk_values[top_indices])][:3]

        explanation_text = "Patient condition is stable. Protective factors are " + ", ".join(
            [translations[features[i]] for i in top_indices]
        )

    
    top_reasons = []
    for i in top_indices:
        top_reasons.append({
            "feature": translations[features[i]],
            "value":   raw_vitals_list[i],
            "impact":  round(float(risk_values[i]), 3),
            "direction": "increases risk" if risk_values[i] > 0 else "reduces risk",
        })

 
    result = {
        "status":      "Risk" if prediction == 1 else "Normal",
        "explanation": explanation_text,
        "risk_factors": top_reasons,
        "warnings":    warnings,
    }

    return json.dumps(result, indent=2)
